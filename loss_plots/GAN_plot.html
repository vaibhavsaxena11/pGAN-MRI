def read_file(filename):
    file1 = open(filename, 'r')
    lines = file1.readlines()
    dloss = []
    gloss = []
    gper = []
    gmse = []
    gfreq = []
    time = []
    for i in range(len(lines)):
        # print(i)
        l = (lines[i][:-3]).split(' ')
        # print(l)
        dloss.append(float(l[3]))
        gloss.append(float(l[5]))
        gper.append(float(l[7]))
        gmse.append(float(l[9]))
        gfreq.append(float(l[11]))
        time.append(float(l[13]))
    file1.close()
    return np.array(dloss), np.array(gloss), np.array(gper), np.array(gmse), np.array(gfreq), np.array(time)
    
def plot_loss(loss, title):
    sns.lineplot(range(1, len(loss)+1), loss)
    plt.ylabel(title)
    plt.xlabel("Step")
    plt.show()
